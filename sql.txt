create table business(
   bid INT NOT NULL AUTO_INCREMENT,
   uid INT ,
   bname VARCHAR(100) ,
   bdescriptions VARCHAR(100) ,
   bdescriptionl VARCHAR(100) ,
   bgender VARCHAR(100) ,
   barea VARCHAR(100) ,
   bcity VARCHAR(100) ,
   baddress VARCHAR(100) ,
   bphone VARCHAR(100) ,
   bsubject VARCHAR(100) ,
   bvisibility VARCHAR(100) ,
   blikes VARCHAR(100) ,
   bviews VARCHAR(100) ,
   createdate datetime ,
   lastlogindate datetime,
	blogo VARCHAR(255) ,
   foreign key ( uid ) REFERENCES  users(userid),
   PRIMARY KEY ( bid )
);

CREATE TABLE reviews (
  reviewid int NOT NULL AUTO_INCREMENT,
  bid int DEFAULT NULL,
  uid int DEFAULT NULL,
   review int,
   reviewtext VARCHAR(100) ,
   reviewdate varchar(100) ,
   foreign key ( uid ) REFERENCES  users(userid),
   foreign key ( bid ) REFERENCES  business(bid),
   PRIMARY KEY ( reviewid )
);
